class Scaler:
    """Standardize the features of an array.
    """
    def __init__(self):
        self._mean = None
        self._std = None

    @property
    def mean(self):
        mean = self._mean
        if mean is None:
            raise ValueError(f'{type(self).__name__} has not been fit')
        return mean

    @property
    def std(self):
        std = self._std
        if std is None:
            raise ValueError(f'{type(self).__name__} has not been fit')
        return std

    def fit(self, data):
        """Fit the scaler to the data.

        Parameters
        ----------
        data : np.ndarray
            The data to fit the scaler to.

        Returns
        -------
        transformed : np.ndarray
            The transformed input data.
        """
        self._mean = data.mean(axis=(0, 1))
        self._std = data.std(axis=(0, 1))
        return self.transform(data)

    def transform(self, data):
        """Scale data based on the fit data.

        Parameters
        ----------
        data : np.ndarray
            The data to transform.

        Returns
        -------
        transformed : np.ndarray
            The transformed input data.
        """
        return (data - self.mean) / self.std


class CappedLoss:
    """A collection of loss functions with a cap.

    Parameters
    ----------
    upper : float
        The upper limit of the loss function.

    Notes
    -----
    This is useful when there is an upper limit to the distance you can be
    wrong by. For example, if a prediction is more than 190 milliseconds away
    from the true click time, it will never change the user's accuracy or score
    so we can set a hard cap.
    """
    def __init__(self, upper):
        self._upper = upper

    def mae(self, true, predicted):
        return keras.backend.mean(
            keras.backend.clip(
                keras.backend.abs(true - predicted),
                0,
                self._upper,
            ),
            axis=-1,
        )

    def mse(self, true, predicted):
        return keras.backend.mean(
            keras.backend.clip(
                keras.backend.square(keras.backend.abs(true - predicted)),
                0,
                self._upper,
            ),
            axis=-1,
        )



class _InnerLSTM:
    """An LSTM model for osu! which trains on windows of (x, y, time) events.

    Parameters
    ----------
    context : int
        The number of leading and trailing hit objects or slider ticks to look
        at.
    hidden_layer_sizes : tuple[int, int]
        The sizes of the hidden layers.
    dropout : float
        The droput ratio.
    activation : str
        The activation function. Must be one of:
        'tanh', 'softplus, 'softsign', 'relu', 'sigmoid', 'hard_sigmoid', or
        'linear'.
    loss : {'mse', 'mae', 'mape', 'male', 'cosine'
        The loss function.
    optimizer : str
        The optimizer to use.
    max_hit_objects : int
        The maximum number of hit objects allows in a beatmap. Data will be
        padded to fit this shape.

    Notes
    -----
    The inner model does not know about the difference between hidden and
    non-hidden plays. :class:`~slider.model.lstm.LSTM` internally
    holds two ``_InnerLSTM``\s which are fitted to the hidden and
    non-hidden plays respectively.
    """
    features = 'x', 'y', 'time'

    _missing_value = 1e9

    def __init__(self,
                 *,
                 context=3,
                 batch_size=32,
                 hidden_layer_sizes=(128, 256),
                 dropout=0.2,
                 activation='tanh',
                 loss='mse',
                 optimizer='adam'):

        self._context = context
        self._window = window = 2 * context + 1

        self._batch_size = batch_size

        self._model = model = keras.models.Sequential([
            keras.layers.LSTM(
                hidden_layer_sizes[0],
                dropout=dropout,
                return_sequences=True,
                input_shape=(
                    window,
                    len(self.features),
                ),
            ),
            keras.layers.LSTM(
                hidden_layer_sizes[1],
                dropout=dropout,
            ),
            keras.layers.Dense(2),
            keras.layers.Activation(activation),
        ])

        model.compile(
            # cap the loss at 190 because that is the maximum time offset
            # which could still be a miss
            loss=getattr(CappedLoss(190), loss),
            optimizer=optimizer,
        )

        self._scaler = Scaler()

    def _extract_features(self,
                          beatmap,
                          *,
                          double_time=False,
                          half_time=False,
                          hard_rock=False):
        """Extract the feature array from a beatmap.

        Parameters
        ----------
        beatmap : Beatmap
            The beatmap to extract features for.
        double_time : bool, optional
            Extract features for double time?
        half_time : bool, optional
            Extract features for half time?
        hard_rock : bool, optional
            Extract features for hard rock?
        """
        hit_objects = beatmap.hit_objects_no_spinners

        # events holds an (x, y, time) tuple for every hit object and slider
        # tick; slider ticks are in ms
        events = []
        append_event = events.append
        extend_events = events.extend

        # hit_object_ixs holds the indices into ``events`` where hit objects
        # appear
        hit_object_ixs = []
        append_hit_object_ixs = hit_object_ixs.append
        for hit_object in hit_objects:
            # mark this index in events as the location of a hit object.
            append_hit_object_ixs(len(events))

            if double_time:
                hit_object = hit_object.double_time
            elif half_time:
                hit_object = hit_object.half_time

            if hard_rock:
                hit_object = hit_object.hard_rock

            position = hit_object.position
            append_event((
                position.x,
                position.y,
                hit_object.time.total_seconds() * 1000,
            ))

            if isinstance(hit_object, Slider):
                # add all the slider ticks
                extend_events(
                    (x, y, time.total_seconds() * 1000)
                    for x, y, time in hit_object.tick_points
                )

        # the padding data for hit objects that are too close to the start
        # or end of a beatmap
        context_padding = np.full((self._context, 3), np.nan)
        # convert the events into a 2d array of shape (len(events), 3)
        event_array = np.array(events)

        # the baseline data to take windows out of
        baseline = np.vstack([
            context_padding,
            event_array,
            context_padding,
        ])

        # convert the hit object indices into a column vector; we add context
        # to account for the padding
        hit_object_ixs_array = (
            np.array(hit_object_ixs) + self._context
        )[:, np.newaxis]

        # get an array of offsets from some base ``ix`` which correspond to a
        # fully populated window; this is a row vector
        context_ix_offsets = np.arange(
            -self._context,
            self._context + 1,
        )

        # broadcast the hit_object_ixs together with the context_ix_offsets
        # to get an indexer which produces a 3d array which is a sequence of
        # windows of events
        window_ixs = hit_object_ixs_array + context_ix_offsets

        # a sequence of windows of events where the time is still absolute
        windows = baseline[window_ixs]

        # slice out the time of each hit object; add a new axis to have a shape
        # of (len(hit_objects), 1)
        hit_object_times = windows[:, self._context, 2, np.newaxis]

        # subtract the hit object time from the event time to make each event's
        # time relative to the current hit object
        windows[:, :, 2] -= hit_object_times

        # fill the nans with a large value to be finite
        windows[np.isnan(windows)] = self._missing_value

        return windows

    def _extract_differences(self, replay):
        """Extract the time and position differences for each hit object.

        Parameters
        ----------
        replay : Replay
            The replay to get differences for.

        Returns
        -------
        Returns
        -------
        differences : np.ndarray
            An array of shape (len(hit_objects), 2) where the first column
            is the time offset in milliseconds and the second column is the
            magnitude of (x, y) error in osu! pixels.
        """
        # get the x, y, and time of each click
        clicks = np.array([
            (
                second.position.x,
                second.position.y,
                second.offset.total_seconds() * 1000,
            )
            for first, second in toolz.sliding_window(2, replay.actions)
            if (second.key1 and not first.key1 or
                second.key2 and not first.key2)
        ])

        double_time = replay.double_time
        half_time = replay.half_time
        hard_rock = replay.hard_rock

        # accumulate the (x, y, start time) of each hit object
        hit_object_coords = []
        append_coord = hit_object_coords.append
        for hit_object in replay.beatmap.hit_objects_no_spinners:
            if double_time:
                hit_object = hit_object.double_time
            elif half_time:
                hit_object = hit_object.half_time

            if hard_rock:
                hit_object = hit_object.hard_rock

            position = hit_object.position
            append_coord((
                position.x,
                position.y,
                hit_object.time.total_seconds() * 1000,
            ))

        # convert the hit object coordinates into an array
        hit_object_coords = np.array(hit_object_coords)

        # get the time of each hit object as a row vector
        hit_object_times = hit_object_coords[:, 2]

        # get the time of each click as a column vector
        click_times = clicks[:, [2]]

        # find the nearest click by taking the absolute difference from
        # every hit object to every click (whose shape is:
        # (len(clicks), len(hit_objects))) and reducing with agrgmin to
        # get the index of the best match for each hit object
        nearest_click_ix = np.abs(
            hit_object_times - click_times,
        ).argmin(axis=0)

        # get the x, y, time of the matched clicks
        matched_clicks = clicks[nearest_click_ix]

        # allocate an empty output
        out = np.empty((len(hit_object_coords), 2), dtype='float64')

        # write the absolute time offset into column 1
        np.abs(
            hit_object_times - matched_clicks[:, 2],
            out=out[:, 0],
        )

        # get the squared distance for the x and y axes
        squared_distance = (
            hit_object_coords[:, :2] - matched_clicks[:, :2]
        ) ** 2

        # write the absolute distance from the circle into column 2
        np.sqrt(
            squared_distance[:, 0] + squared_distance[:, 1],
            out=out[:, 1],
        )

        return out

    def fit(self, replays, *, verbose=False):
        extract_features = self._extract_features
        extract_differences = self._extract_differences
        model = self._model

        features = []
        append_features = features.append
        labels = []
        append_labels = labels.append
        for n, replay in enumerate(replays):
            if verbose:
                print(f'{n:4}: {replay!r}')

            append_features(extract_features(
                replay.beatmap,
                double_time=replay.double_time,
                half_time=replay.half_time,
                hard_rock=replay.hard_rock,
            ))
            append_labels(extract_differences(replay))

        model.fit(
            self._scaler.fit(np.concatenate(features)),
            np.concatenate(labels),
            verbose=int(bool(verbose)),
            batch_size=self._batch_size,
        )

    def predict_differences(self,
                            beatmap,
                            *,
                            double_time=False,
                            half_time=False,
                            hard_rock=False):
        """Predict the time and position differences for each circle.

        Parameters
        ----------
        beatmap : Beatmap
            The beatmap to predict.
        double_time : bool, optional
            Predict double time offsets.
        half_time : bool, optional
            Predict half time offsets.
        hard_rock : bool, optional
            Predict hard_rock offsets.

        Returns
        -------
        predicted_differences : np.ndarray
            An array of shape (len(hit_objects), 2) where the first column
            is the predicted time offset in milliseconds and the second
            column is the predicted magnitude of (x, y) error in osu! pixels.
        """
        return self._model.predict(
            self._scaler.transform(
                self._extract_features(
                    beatmap,
                    double_time=double_time,
                    half_time=half_time,
                    hard_rock=hard_rock,
                ),
            ),
        )

    def _fit_truncnorm(self, array):
        """Fit a truncated normal distribution to the array. The distribution
        is truncated to ``[0, self._missing_value]``

        Parameters
        ----------
        array : np.ndarray
            The data to fit.
        lower : float
            The lower bound.
        upper : float
            The upper bound.

        Returns
        -------
        distribution : scipy.stats.truncnorm
            A frozen truncated normal instance.
        """
        lower = 0
        upper = self._missing_value
        mean = array.mean()
        std = array.std()
        return scipy.stats.truncnorm(
            (lower - mean) / std,
            (upper - mean) / std,
            loc=mean,
            scale=std,
        )

    def _predict_accuracy(self,
                          beatmap,
                          *,
                          double_time=False,
                          half_time=False,
                          hard_rock=False):
        """Predict the user's accuracy on the beatmap with the given mods.

        Parameters
        ----------
        beatmap : Beatmap
            The beatmap to predict.
        double_time : bool, optional
            Predict double time offsets.
        half_time : bool, optional
            Predict half time offsets.
        hard_rock : bool, optional
            Predict hard_rock offsets.

        Returns
        -------
        predicted_accuracy : float
            A scalar value for the user's predicted accuracy.
        """
        diffs = self.predict_differences(
            beatmap,
            double_time=double_time,
            half_time=half_time,
            hard_rock=hard_rock,
        )

        aim, accuracy = diffs.T

        aim_distribution = self._fit_truncnorm(aim)
        accuracy_distribution = self._fit_truncnorm(accuracy)

        hit_object_count = len(beatmap.hit_objects)
        sampled_aim = aim_distribution.rvs(hit_object_count)
        sampled_accuracy = accuracy_distribution.rvs(hit_object_count)

        hit_windows = np.array(
            od_to_ms(
                beatmap.od(
                    hard_rock=hard_rock,
                    double_time=double_time,
                    half_time=half_time,
                ),
            ),
        )[:, np.newaxis]

        # compare the sampled accuracy to the hit windows to produce a boolean
        # array of shape (3, hit_object_count) where each element is if
        # the sampled accuracy falls within the row's hit threshold
        hit_window_mask = sampled_accuracy <= hit_windows

        accuracy_per_hit_object = np.select(
            [
                # if the aim is > the circle radius then we missed and score 0
                sampled_aim > circle_radius(beatmap.cs(hard_rock=hard_rock)),
                # the sampled accuracy falls within the 300 point threshold
                hit_window_mask[0],
                # the sampled accuracy falls within the 100 point threshold
                hit_window_mask[1],
                # the sampled accuracy falls within the 50 point threshold
                hit_window_mask[2],
            ],
            [
                0,
                1,
                1 / 3,
                1 / 6,
            ],
            # if we don't fall in the 300, 100, or 50 hit thresholds then we
            # have a miss which scores 0 points.
            default=0,
        )

        # return the mean accuracy
        return accuracy_per_hit_object.mean()

    def predict_one(self, beatmap, mods):
        return self._predict_accuracy(
            beatmap,
            double_time=mods['double_time'],
            half_time=mods['half_time'],
            hard_rock=mods['hard_rock'],
        )

    def save_path(self, path):
        self._model.save(path)

    @classmethod
    def load_path(cls, path):
        self = cls()
        self._model = keras.models.load_model(path)
        return self



[docs]class LSTM(OsuModel):
    """An LSTM model for osu! which trains on windows of (x, y, time) events.

    Parameters
    ----------
    context : int
        The number of leading and trailing hit objects or slider ticks to look
        at.
    hidden_layer_sizes : tuple[int, int]
        The sizes of the hidden layers.
    dropout : float
        The droput ratio.
    activation : str
        The activation function.
    loss : str
        The loss function.
    optimizer : str
        The optimizer to use.
    max_hit_objects : int
        The maximum number of hit objects allows in a beatmap. Data will be
        padded to fit this shape.
    """
    def __init__(self, *args, **kwargs):
        self._hidden_model = _InnerLSTM(*args, **kwargs)
        self._non_hidden_model = _InnerLSTM(*args, **kwargs)

    def save_path(self, path):
        os.makedirs(path, exist_ok=True)
        self._hidden_model.save_path(os.path.join(path, 'hidden'))
        self._non_hidden_model.save_path(os.path.join(path, 'non-hidden'))

    @classmethod
    def load_path(cls, path):
        self = cls()
        self._hidden_model = _InnerLSTM.load_path(
            os.path.join(path, 'hidden'),
        )
        self._non_hidden_model = _InnerLSTM.load_path(
            os.path.join(path, 'non-hidden'),
        )
        return self

    def fit(self, replays, *, verbose=False):
        hidden, non_hidden = dichotomize(lambda replay: replay.hidden, replays)

        if verbose:
            print('fitting the hidden replays')
            self._hidden_model.fit(hidden, verbose=verbose)

        if verbose:
            print('fitting the non-hidden replays')
        self._non_hidden_model.fit(non_hidden, verbose=verbose)

    def predict(self, beatmaps_and_mods):
        hidden_model_predict = self._hidden_model.predict_one
        non_hidden_model_predict = self._non_hidden_model.predict_one

        return np.array([
            (
                hidden_model_predict
                if mods['hidden'] else
                non_hidden_model_predict
            )(beatmap, mods)
            for beatmap, mods in beatmaps_and_mods
        ])


[docs]    def predict_differences(self, beatmaps_and_mods):
        """Low level predictions for a sequence of beatmaps and mods.

        Parameters
        ----------
        beatmaps_and_mods : iterable[(Beatmap, dict)]
            The maps and mods to predict.

        Returns
        -------
        differences : list[np.ndarray[float]]
            For each beatmap and mod combination, an array of shape
            ``(len(hit_objects), 2)`` where the first column holds the
            predicted absolute click offset from the center of the circle and
            the second column holds the absolute time offset for the predicted
            click. The row indices match the order the beatmap and mod
            combinations are passed.

        Notes
        -----
        This method is specific to the :class:`~slider.model.lstm.LSTM` model,
        it may not be available on other model types.
        """
        hidden_model_predict = self._hidden_model.predict_differences
        non_hidden_model_predict = self._non_hidden_model.predict_differences

        return [
            (
                hidden_model_predict
                if mods['hidden'] else
                non_hidden_model_predict
            )(
                beatmap,
                double_time=mods['double_time'],
                half_time=mods['half_time'],
                hard_rock=mods['hard_rock'],
            )
            for beatmap, mods in beatmaps_and_mods
        ]



[docs]    def predict_beatmap_differences(self, beatmap, *mods, **mods_scalar):
        """Predict the user's accuracy for the given beatmap.

        Parameters
        ----------
        beatmap : Beatmap
            The map to predict per hit object differences of.
        *mods
            A sequence of mod dictionaries to predict per hit object
            differences for.
        **mods_dict
            Mods to predict per hit object differences for.

        Returns
        -------
        differences :list[np.ndarray[float]]
            For each mod combination, an array of shape
            ``(len(hit_objects), 2)`` where the first column is the user's
            expected absolute difference for the click position and the second
            column is the expected absolute difference click time.

        Notes
        -----
        This method is specific to the :class:`~slider.model.lstm.LSTM` model,
        it may not be available on other model types.
        """
        for mod_name in 'hidden', 'hard_rock', 'half_time', 'double_time':
            mods_scalar.setdefault(mod_name, False)

        return self.predict_differences([
            (beatmap, ms) for ms in chain(mods, [mods_scalar])
        ])
