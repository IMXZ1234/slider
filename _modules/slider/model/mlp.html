import pickle

from sklearn.neural_network import MLPRegressor as _MLPRegressor
from sklearn.preprocessing import StandardScaler

from .model import OsuModel
from .manual_features import extract_feature_array



[docs]class MLPRegressor(OsuModel):
    """An osu! aware MLPRegressor.

    Parameters
    ----------
    **kwargs
        Forwarded to ``sklearn.neural_network.MLPRegressor``.
    """
    def __init__(self,
                 alpha=0.009,
                 solver='lbfgs',
                 activation='tanh',
                 hidden_layer_sizes=(54, 199, 66),
                 max_iter=1000,
                 tol=1e-9,
                 warm_start=True,
                 **kwargs):
        self._scaler = StandardScaler()
        self._model = _MLPRegressor(
            alpha=alpha,
            solver=solver,
            activation=activation,
            hidden_layer_sizes=hidden_layer_sizes,
            max_iter=max_iter,
            tol=tol,
            warm_start=warm_start,
            **kwargs,
        )

    def save_path(self, path):
        with open(path, 'wb') as f:
            pickle.dump(self, f)

    @classmethod
    def load_path(self, path):
        with open(path, 'rb') as f:
            return pickle.load(f)

    def fit(self, replays, *, verbose=False):
        train, labels = extract_feature_array(replays)
        train = self._scaler.fit_transform(train, labels)
        return self._model.fit(train, labels)

    def predict(self, beatmaps_and_mods):
        features = extract_feature_array(beatmaps_and_mods)
        return self._model.predict(self._scaler.transform(features))
